PHASE 2 Data Ingestion Pipeline
5
Write the data ingestion script (ingest.py)
Load → chunk → embed → store in ChromaDB
▶
# ingest.py from langchain_community.document_loaders import ( PyPDFLoader, DirectoryLoader ) from langchain.text_splitter import RecursiveCharacterTextSplitter from langchain_community.embeddings import HuggingFaceEmbeddings from langchain_community.vectorstores import Chroma loader = DirectoryLoader("data/", glob="**/*.pdf", loader_cls=PyPDFLoader, show_progress=True) documents = loader.load() print(f"Loaded {len(documents)} pages") splitter = RecursiveCharacterTextSplitter( chunk_size=500, chunk_overlap=50, separators=["\n\n", "\n", ".", " "] ) chunks = splitter.split_documents(documents) print(f"Created {len(chunks)} chunks") embeddings = HuggingFaceEmbeddings( model_name="sentence-transformers/all-MiniLM-L6-v2" ) vectordb = Chroma.from_documents( documents=chunks, embedding=embeddings, persist_directory="./vectorstore" ) vectordb.persist() print("Vector store created!")
First run downloads the embedding model (~90MB). The vectorstore/ folder stores your indexed data permanently.
6
Build the chatbot (chatbot.py)
Query → retrieve chunks → LLM generates answer
▶
# chatbot.py from langchain_community.embeddings import HuggingFaceEmbeddings from langchain_community.vectorstores import Chroma from langchain_ollama import ChatOllama from langchain.chains import RetrievalQA from langchain.prompts import PromptTemplate embeddings = HuggingFaceEmbeddings( model_name="sentence-transformers/all-MiniLM-L6-v2" ) vectordb = Chroma(persist_directory="./vectorstore", embedding_function=embeddings) retriever = vectordb.as_retriever( search_type="similarity", search_kwargs={"k": 4} ) PROMPT = PromptTemplate( input_variables=["context", "question"], template="""You are a helpful assistant. Use ONLY the context below. If the answer is not in context, say "I don't know based on provided data." Context: {context} Question: {question} Answer:""" ) llm = ChatOllama(model="llama3.2", temperature=0) qa_chain = RetrievalQA.from_chain_type( llm=llm, chain_type="stuff", retriever=retriever, chain_type_kwargs={"prompt": PROMPT}, return_source_documents=True ) print("RAG Chatbot ready. Type 'quit' to exit.\n") while True: query = input("You: ") if query.lower() == "quit": break result = qa_chain.invoke({"query": query}) print(f"\nBot: {result['result']}") for doc in result['source_documents']: print(f" Source: {doc.metadata.get('source','?')}") print()

PART A Wrap your chatbot as a REST API
1
Create FastAPI server (api.py)
Expose your RAG chatbot as an HTTP endpoint
▶
# api.py from fastapi import FastAPI, HTTPException from fastapi.middleware.cors import CORSMiddleware from pydantic import BaseModel from typing import Optional import uvicorn, uuid, time from langchain_community.embeddings import HuggingFaceEmbeddings from langchain_community.vectorstores import Chroma from langchain_ollama import ChatOllama from langchain.chains import RetrievalQA from langchain.prompts import PromptTemplate app = FastAPI(title="RAG Chatbot API", version="1.0") app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"]) @app.on_event("startup") async def startup(): global qa_chain embeddings = HuggingFaceEmbeddings( model_name="sentence-transformers/all-MiniLM-L6-v2") vectordb = Chroma(persist_directory="./vectorstore", embedding_function=embeddings) retriever = vectordb.as_retriever(search_kwargs={"k": 4}) llm = ChatOllama(model="llama3.2", temperature=0) PROMPT = PromptTemplate( input_variables=["context","question"], template="Context: {context}\n\nQuestion: {question}\nAnswer:" ) qa_chain = RetrievalQA.from_chain_type( llm=llm, retriever=retriever, chain_type_kwargs={"prompt": PROMPT}, return_source_documents=True ) class QueryRequest(BaseModel): question: str session_id: Optional[str] = None class QueryResponse(BaseModel): answer: str sources: list session_id: str processing_time_ms: int @app.post("/api/chat", response_model=QueryResponse) async def chat(req: QueryRequest): if not req.question.strip(): raise HTTPException(400, "Question cannot be empty") start = time.time() result = qa_chain.invoke({"query": req.question}) elapsed = int((time.time() - start) * 1000) sources = [{ "file": doc.metadata.get("source","?"), "page": doc.metadata.get("page","?"), "content": doc.page_content[:200] } for doc in result["source_documents"]] return QueryResponse( answer=result["result"], sources=sources, session_id=req.session_id or str(uuid.uuid4()), processing_time_ms=elapsed ) @app.get("/health") async def health(): return {"status": "ok"} if __name__ == "__main__": uvicorn.run(app, host="0.0.0.0", port=8000, reload=True)
FastAPI auto-generates interactive docs at http://localhost:8000/docs

3
Full Python integration with data.gov.in
Fetch, filter, paginate — complete code
▶
# datagov_client.py import requests, json, os from dotenv import load_dotenv load_dotenv() API_KEY = os.getenv("DATAGOV_API_KEY") BASE_URL = "https://api.data.gov.in/resource" def fetch_data(resource_id, filters=None, limit=100, offset=0): params = { "api-key": API_KEY, "format": "json", "limit": limit, "offset": offset } if filters: for key, value in filters.items(): params[f"filters[{key}]"] = value r = requests.get(f"{BASE_URL}/{resource_id}", params=params, timeout=30) r.raise_for_status() return r.json() def fetch_all_pages(resource_id, filters=None, page_size=500): all_records, offset = [], 0 while True: data = fetch_data(resource_id, filters, page_size, offset) records = data.get("records", []) total = int(data.get("total", 0)) all_records.extend(records) offset += len(records) print(f"Fetched {offset}/{total} records...") if offset >= total or not records: break return all_records # Usage data = fetch_data("YOUR_RESOURCE_ID", limit=10) print("Total:", data.get("total")) print("Sample:", json.dumps(data["records"][0], indent=2)) # Filter by state gujarat = fetch_data("RESOURCE_ID", filters={"state": "Gujarat"}) # Fetch all and save all_data = fetch_all_pages("RESOURCE_ID") with open("data/govdata.json","w") as f: json.dump(all_data, f, indent=2)

PRODUCTION CHECKLIST
5
Before going to production — complete checklist
Security, performance, monitoring, deployment
▶

Security

  • API keys in .env, not in code
  • Rate limiting per IP/key
  • CORS restricted to your domain
  • HTTPS via nginx + SSL
  • Input validation & max length

Performance

  • Load RAG chain once at startup
  • Add response caching (Redis)
  • Async endpoints (FastAPI async)
  • Optimise chunk size (test 300–800)
  • Use GPU if available

Monitoring

  • Log every query + response time
  • Track source relevance scores
  • Add /metrics (Prometheus)
  • Set up error alerts

Deployment

  • Docker + docker-compose
  • nginx as reverse proxy
  • Systemd service (auto-restart)
  • Backup vectorstore/ regularly
